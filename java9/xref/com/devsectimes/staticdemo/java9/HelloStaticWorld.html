17  public int add(String... args) {
18  
19      List<Integer> arglist = new ArrayList<>();
20  
21      for(int i=0;i<args.length;i++) {
22        arglist.add(Integer.parseInt(args[i]));
23      }
24  
25      int total=0;
26  
27      Iterator<Integer> i=arglist.iterator();
28  
29      while(i.hasNext()) {
30        total+=i.next();
31      }
32  
33      return total;
