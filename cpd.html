public int add(String... args) {

    List<Integer> arglist = new ArrayList<>();

    for(int i=0;i<args.length;i++) {
      arglist.add(Integer.parseInt(args[i]));
    }

    int total=0;

    Iterator<Integer> i=arglist.iterator();

    while(i.hasNext()) {
      total+=i.next();
    }

    return total;
  }
